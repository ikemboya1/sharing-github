# titanic_analysis.py

# Import libraries
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Step 1: Load Data
df = sns.load_dataset('titanic')

# Step 2: Initial Exploration
print("\n=== Initial Data Exploration ===")
print("First 5 rows:")
print(df.head())
print("\nMissing values before cleaning:")
print(df.isnull().sum())

# Step 3: Data Cleaning
# Drop 'deck' (too many missing values) and rows with missing 'embarked'
df_clean = df.drop(columns=['deck', 'alive', 'alone'])
df_clean = df_clean.dropna(subset=['embarked'])

# Fill missing 'age' with median and convert 'sex' to numerical
df_clean['age'] = df_clean['age'].fillna(df_clean['age'].median())
df_clean['sex'] = df_clean['sex'].map({'male': 0, 'female': 1})

# Verify cleaning
print("\n=== After Cleaning ===")
print("Missing values after cleaning:")
print(df_clean.isnull().sum())

# Step 4: Visualization
plt.figure(figsize=(12, 10))

# Plot 1: Survival by Class
plt.subplot(2, 2, 1)
sns.barplot(data=df_clean, x='class', y='survived', ci=None)
plt.title('Survival Rate by Class')

# Plot 2: Age Distribution by Survival
plt.subplot(2, 2, 2)
sns.histplot(data=df_clean, x='age', hue='survived', kde=True, bins=20)
plt.title('Age Distribution by Survival')

# Plot 3: Fare vs. Age
plt.subplot(2, 2, 3)
sns.scatterplot(data=df_clean, x='age', y='fare', hue='survived', alpha=0.6)
plt.title('Fare vs. Age (Survival)')

# Plot 4: Survival by Gender
plt.subplot(2, 2, 4)
sns.countplot(data=df_clean, x='sex', hue='survived')
plt.title('Survival Count by Gender')
plt.xticks([0, 1], ['Male', 'Female'])

plt.tight_layout()
plt.show()

# Step 5: Final Report
report = f"""
=== Titanic Dataset Analysis Report ===
Total Passengers: {len(df_clean)}
Survival Rate: {df_clean['survived'].mean() * 100:.1f}%

Key Findings:
1. Class Impact: 
   - 1st Class Survival: {df_clean[df_clean['class'] == 'First']['survived'].mean() * 100:.1f}%
   - 3rd Class Survival: {df_clean[df_clean['class'] == 'Third']['survived'].mean() * 100:.1f}%

2. Gender Impact:
   - Female Survival: {df_clean[df_clean['sex'] == 1]['survived'].mean() * 100:.1f}%
   - Male Survival: {df_clean[df_clean['sex'] == 0]['survived'].mean() * 100:.1f}%

3. Age Analysis:
   - Avg Age: {df_clean['age'].mean():.1f} years
   - Max Age: {df_clean['age'].max()} years
   - Min Age: {df_clean['age'].min()} years

4. Fare Analysis:
   - Avg Fare: ${df_clean['fare'].mean():.2f}
   - Max Fare: ${df_clean['fare'].max():.2f}
"""

print(report)

# Optional: Save cleaned data
df_clean.to_csv('cleaned_titanic.csv', index=False)
print("\nCleaned data saved to 'cleaned_titanic.csv'.")
